class Main {
    public static void main(String[] args) {
        int [] a={1,3,3,1,5,8,3,3};
        int k=3;
        int l=0,r=k-1,sum=0;
        int max_sum=0;
        for (int i = 0; i < k; i++) {
            sum += a[i];
        }

        max_sum = sum;
        while(r<a.length-1){
            sum=sum-a[l++];
            sum=sum+a[++r];
            max_sum=Math.max(max_sum,sum);
        }
        System.out.println(max_sum);
    }
}
