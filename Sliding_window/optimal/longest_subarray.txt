class Main {
    public static void main(String[] args) {
        int[] a = {5, 2, 4, 1, 1};
int k = 5;


        int r=0,l=0,sum=0,max=0;
        while(r<a.length){
            sum=sum+a[r];
            if(sum>k){
                sum=sum-a[l];
                l++;
            }
                max=Math.max(max,r-l+1);
                r++;
            
        }
        System.out.println(max);
    }
}
