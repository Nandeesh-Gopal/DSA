class Main {
    public static void main(String[] args) {
        int [] a={1,3,3,1,5,8,3,3};
        int k=10;
        int r=0,l=0,sum=0,max=0;
        while(r<a.length){
            sum=sum+a[r];
            while(sum>k){
                sum=sum-a[l];
                l++;
            }
            if(sum<=k){
                max=Math.max(max,r-l+1);
                r++;
            }
        }
        System.out.println(max);
    }
}
