class Main {
    public static void main(String[] args) {
        int [] a={1,3,3,1,5,8,3,3};
        int k=10;
        int max=0;
        for(int i=0;i<a.length;i++){
            int sum=0;
            for(int j=i;j<a.length;j++){
                sum=sum+a[j];
                if(sum<k){
                    max=Math.max(max,j-i+1);
                }
            }
        }
        System.out.println(max);
        
    }
}
