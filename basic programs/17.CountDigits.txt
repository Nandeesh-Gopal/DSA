import java.util.Scanner;
class Main {
    public static void main(String[] args) {
        Scanner sc =new Scanner(System.in);
        System.out.print("Enter: ");
        int n=sc.nextInt();
        int c=0;
        if(n==0){
            c=1;
        }
        else{
        if(n<0){
            n= n*-1;
        }
        while(n>0){
            n=n/10;
            c++;
        }}
        System.out.println("No of digit is "+c);
    }
}
