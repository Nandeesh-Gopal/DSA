class Main {
    public static void main(String[] args) {
        int [] a={1,2,3,4,5,6,7,8,9,0};
        int e=0,c=0;
        for(int i=0;i<a.length;i++){
            if(a[i]%2==0){
                e++;
            }
            else{
                c++;
            }
        }
        System.out.println(e+" "+c);
    }
}
