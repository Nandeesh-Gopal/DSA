import java.util.Scanner;
class Main {
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        System.out.println("Enter Number:");
        int n=sc.nextInt();
        int s=0;
        for(int i=1;i<n+1;i++){
            System.out.println(i);
            s=s+i;
        }
        System.out.println("Sum: "+ s);
    }
}
