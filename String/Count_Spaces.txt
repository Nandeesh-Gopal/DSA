
class Main {
    public static void main(String[] args) {
        String s="madam 123 4567 ";
        System.out.println(count_spaces(s));
    }
    public static int count_spaces(String s){
        int count =0;
        for(int i=0;i<s.length();i++){
            if(Character.isSpaceChar(s.charAt(i))){
                count++;
            }
        }
        return count;
    }
}
