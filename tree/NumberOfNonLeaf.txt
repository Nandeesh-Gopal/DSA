import java.util.*;
public class Main{
    public static void main(String[] args){
        Node n=new Node(5);
        n.left=new Node(1);
        n.right=new Node(4);
        n.right.left=new Node(3);
        n.right.right=new Node(6);

        System.out.println("Number of node except leaf in the tree: "+NonLeaf(n));
    }
    public static int NonLeaf(Node node){
        if(node==null)return 0;
        if(node.left==null && node.right==null){
            return 0;
        }
        return 1+NonLeaf(node.left)+NonLeaf(node.right);
    }
    
    
}class Node{
    int data;
    Node left;
    Node right;
    Node(int data){
        this.data=data;
    }
}
