import java.util.*;
public class Main{
    public static void main(String[] args){
        Node n=new Node(5);
        n.left=new Node(1);
        n.right=new Node(3456);
        n.right.left=new Node(-3);
        n.right.right=new Node(9);
        
        System.out.println("\nMaximum in a tree: "+findmax(n));
        System.out.println("Minimum in a tree: "+findmin(n));
    }
    public static int findmax(Node node){
        if(node ==null) return Integer.MIN_VALUE;
        int leftmax=findmax(node.left);
        int rightmax=findmax(node.right);
        
        return Math.max(node.data,Math.max(leftmax,rightmax));
        
    }
    public static int findmin(Node node){
        if(node==null) return Integer.MAX_VALUE;
        int leftmin=findmin(node.left);
        int rightmin=findmin(node.right);
        return Math.min(node.data,Math.min(leftmin,rightmin));
    }
}class Node{
    int data;
    Node left;
    Node right;
    Node(int data){
        this.data=data;
    }
}
