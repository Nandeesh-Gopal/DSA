import java.util.*;
class Solution {
    static int evenlyDivides(int n) {
        List<Integer> l=new ArrayList<>();
        int ori=n;
        int count=0;
        while(n!=0){
            if(n%10 !=0){
                l.add(n%10);
            }
            n=n/10;
        }
        for(int i=0;i<l.size();i++){
            if(ori%l.get(i)==0){
                count++;
            }
        }
        return count;
    }
}
