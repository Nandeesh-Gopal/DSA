class Main {
    public static void main(String[] args) {
        int[]a={1,2,3,5,6,8};
        System.out.println(missing_number(a));
    }
    public static int missing_number(int[] a){
        for(int i=1;i<a.length;i++){
            if((a[i]-a[i-1])!=1){
                return i+1;
            }
        }
        return -1;
    }
    
}
