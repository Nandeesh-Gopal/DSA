class Main {
    public static void main(String[] args) {
        int[]a={1,2,3,4,5,6,7,8};
        int[]b=reverse(a);
        for(int i=0;i<b.length;i++){
            System.out.print(b[i]+" ");
        }
    }
    public static int[] reverse(int[] a){
        int l=0;
        int r=a.length-1;
        while(l<r){
            int temp=a[l];
            a[l]=a[r];
            a[r]=temp;
            l++;
            r--;
        }
        return a;
    }
}
